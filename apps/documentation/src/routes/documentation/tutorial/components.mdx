# Components

## Creating a stand-alone component

Letâ€™s then create the button needed for displaying the list of PokÃ©mon.

Create the following file `src/components/PokemonLink.tsx` and fill the content with:

```tsx
// src/components/PokemonLink.tsx
import { Link } from 'tuono'

interface Pokemon {
  name: string
}

export default function PokemonLink({
  pokemon,
  id,
}: {
  pokemon: Pokemon
  id: number
}): JSX.Element {
  return (
    <Link href={`/pokemons/${pokemon.name}`}>
      {pokemon.name}
      <img
        src={`https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${id}.png`}
      />
    </Link>
  )
}
```
